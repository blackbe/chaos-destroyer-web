### 1. `responsiveTester.js`
This script uses Puppeteer to take screenshots of the HugBack app at various breakpoints.

```javascript
// /Users/benblack/.openclaw/workspace/nightly-builds/outputs/2026-03-17-hugback-24/puppeteer/responsiveTester.js
const puppeteer = require('puppeteer');

const takeScreenshot = async (url, name) => {
    const browser = await puppeteer.launch();
    const page = await browser.newPage();

    await page.goto(url);
    await page.setViewport({ width: 375, height: 812 }); // iPhone X
    await page.screenshot({ path: `./screenshots/${name}-iphone-x.png` });

    await page.setViewport({ width: 768, height: 1024 }); // iPad
    await page.screenshot({ path: `./screenshots/${name}-ipad.png` });

    await page.setViewport({ width: 1024, height: 768 }); // iPad Landscape
    await page.screenshot({ path: `./screenshots/${name}-ipad-landscape.png` });

    await page.setViewport({ width: 1440, height: 900 }); // Desktop
    await page.screenshot({ path: `./screenshots/${name}-desktop.png` });

    await page.setViewport({ width: 1920, height: 1080 }); // Full HD
    await page.screenshot({ path: `./screenshots/${name}-fullhd.png` });

    await browser.close();
};

const run = async () => {
    const appUrl = 'https://your-hugback-app-url.com'; // Replace with actual URL
    const pages = ['/', '/login', '/home', '/matches', '/chat', '/profile', '/hugboard', '/storywall'];

    for (const page of pages) {
        await takeScreenshot(`${appUrl}${page}`, page.replace('/', ''));
    }
};

run().catch(console.error);
```
